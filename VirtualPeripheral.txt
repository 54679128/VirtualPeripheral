---@param slot integer
---@return {name: string, count: integer, nbt: string}|nil
function Inventory:getItemDetail(slot) end

---@param slot integer
---@return integer
function Inventory:getItemLimit(slot) end

---@param toName string
---@param fromSlot integer
---@param limit? integer
---@param toSlot? integer
function Inventory:pushItems(toName, fromSlot, limit, toSlot) end

---@param fromName string

---@param limit? integer

-- ==================== LocalNet 子模块（lib.LocalNet） ====================

---@class VirtualPeripheral_LocalNetModule
local LocalNetModule = {}

function LocalNetModule.reset() end

---@param name string

---@param name2 string
---@return integer|nil
